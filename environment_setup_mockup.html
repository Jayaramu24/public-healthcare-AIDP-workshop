Task 7

Create the GOLD_XX schema for claims processing

Admin step
SQL shown in the main working area
  1. Inside AI Lakehouse Database Actions, open the SQL worksheet as the admin user.
  2. Create the workshop gold schema user and grant the privileges needed for the claims processing star schema.
  3. Use this schema as the target for the participant load steps in the later labs.
SQL Worksheet Claims processing gold schema setup 
CREATE USER GOLD_HLTH IDENTIFIED BY "Welcome_12345";

GRANT CONNECT, RESOURCE TO GOLD_HLTH;
GRANT UNLIMITED TABLESPACE TO GOLD_HLTH;

ALTER USER GOLD_HLTH DEFAULT TABLESPACE DATA;

GRANT CREATE SESSION TO GOLD_HLTH;
GRANT CREATE TABLE TO GOLD_HLTH;
GRANT CREATE VIEW TO GOLD_HLTH;
GRANT CREATE SEQUENCE TO GOLD_HLTH;
GRANT CREATE PROCEDURE TO GOLD_HLTH;

-- Claims processing gold objects
CREATE TABLE GOLD_HLTH.MPHA_DIM_DATE (
  DATE_KEY NUMBER PRIMARY KEY,
  FULL_DATE DATE,
  CALENDAR_YEAR NUMBER,
  CALENDAR_QUARTER NUMBER,
  MONTH_NUMBER NUMBER,
  MONTH_NAME VARCHAR2(20),
  WEEK_START_DATE DATE,
  DAY_OF_WEEK VARCHAR2(12),
  IS_WEEK_START CHAR(1)
);

CREATE TABLE GOLD_HLTH.MPHA_DIM_MEMBER (
  MEMBER_KEY NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  MEMBER_ID VARCHAR2(40),
  MEMBER_NAME VARCHAR2(120),
  GENDER VARCHAR2(20),
  DATE_OF_BIRTH DATE,
  AGE_BAND VARCHAR2(30),
  DISTRICT_NAME VARCHAR2(80),
  ENROLLMENT_CHANNEL VARCHAR2(40),
  COVERAGE_START_DATE DATE,
  COVERAGE_END_DATE DATE
);

CREATE TABLE GOLD_HLTH.MPHA_DIM_PROVIDER (
  PROVIDER_KEY NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  PROVIDER_ID VARCHAR2(40),
  PROVIDER_NAME VARCHAR2(140),
  PROVIDER_TYPE VARCHAR2(40),
  SPECIALTY_GROUP VARCHAR2(80),
  DISTRICT_NAME VARCHAR2(80),
  ACCREDITATION_STATUS VARCHAR2(40),
  ACCREDITATION_SCORE NUMBER(10,2)
);

CREATE TABLE GOLD_HLTH.MPHA_DIM_CLAIM_STATUS (
  CLAIM_STATUS_KEY NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  CLAIM_STATUS_CODE VARCHAR2(20),
  CLAIM_STATUS_DESC VARCHAR2(80),
  DENIAL_FLAG CHAR(1)
);

CREATE TABLE GOLD_HLTH.MPHA_FACT_CLAIMS (
  CLAIM_KEY NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  CLAIM_ID VARCHAR2(40),
  DATE_KEY NUMBER REFERENCES GOLD_HLTH.MPHA_DIM_DATE(DATE_KEY),
  MEMBER_KEY NUMBER REFERENCES GOLD_HLTH.MPHA_DIM_MEMBER(MEMBER_KEY),
  PROVIDER_KEY NUMBER REFERENCES GOLD_HLTH.MPHA_DIM_PROVIDER(PROVIDER_KEY),
  CLAIM_STATUS_KEY NUMBER REFERENCES GOLD_HLTH.MPHA_DIM_CLAIM_STATUS(CLAIM_STATUS_KEY),
  CLAIM_TYPE VARCHAR2(40),
  DIAGNOSIS_CATEGORY VARCHAR2(80),
  BILLED_AMOUNT NUMBER(14,2),
  APPROVED_AMOUNT NUMBER(14,2),
  DISBURSED_AMOUNT NUMBER(14,2),
  DENIED_AMOUNT NUMBER(14,2),
  LENGTH_OF_STAY_DAYS NUMBER,
  PROCESSING_DAYS NUMBER,
  CLAIM_COUNT NUMBER
);
This gives the admin a concrete schema creation moment on screen even though the source lab does not provide a separate screenshot sequence for Task 7.
